def tokenizer_nltk(text):
    return nltk.tokenize.sent_tokenize(text)

def remove_punctuation(text):
    return ''.join([item for item in text if item not in string.punctuation])


def remove_stopwords(text):
    if len(text) == 0:
        return text
    return [item for item in text[0].split() if item not in ENGLISH_STOPWORDS]

def contractions_fix(text):
    return contractions.fix(text)

def lemmatize_stemming_option(text, option='stem'):
    if len(text) == 0:
        return text
    text = text[0].split()
    if option != 'stem':
        return [lemmatizer.lemmatize(item) for item in text]
    return [stemmer.stem(item) for item in text]
